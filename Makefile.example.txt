db-backup: ## Sauvegarder la base de données
	@echo "$(YELLOW)💾 Sauvegarde de la base...$(NC)"
	@mkdir -p backups
	mongodump --uri='mongodb://admin:adminpassword@localhost:27017/securite?authSource=admin' --out=backups/backup-$$(date +%Y%m%d-%H%M%S)
	@echo "$(GREEN)✅ Sauvegarde dans backups/$(NC)"

db-restore: ## Restaurer une sauvegarde (usage: make db-restore BACKUP=backups/backup-xxx)
	mongorestore --uri='mongodb://admin:adminpassword@localhost:27017/?authSource=admin' --drop $(BACKUP)

vps-setup: ## Installer MongoDB8+Redis+Nginx+Certbot+PM2+Node24 sur le VPS (première fois)
	@echo "$(YELLOW)🔧 Installation complète de l'environnement VPS...$(NC)"
	@scp scripts/vps-setup.sh $(VPS_USER)@$(VPS_HOST):$(VPS_DIR)/scripts/vps-setup.sh
	@ssh -t $(VPS_USER)@$(VPS_HOST) "sudo bash $(VPS_DIR)/scripts/vps-setup.sh"
	@echo "$(GREEN)✅ VPS configuré$(NC)"

vps-nginx: ## Déployer la config Nginx + GeoIP2 sur le VPS
	@echo "$(YELLOW)🌐 Installation de la configuration Nginx...$(NC)"
	@ssh $(VPS_USER)@$(VPS_HOST) "mkdir -p $(VPS_DIR)/nginx"
	@scp nginx/nginx-main.conf                  $(VPS_USER)@$(VPS_HOST):$(VPS_DIR)/nginx/nginx-main.conf
	@scp nginx/server.a3s-securite.com.conf     $(VPS_USER)@$(VPS_HOST):$(VPS_DIR)/nginx/server.a3s-securite.com.conf
	@scp scripts/vps-nginx-install.sh           $(VPS_USER)@$(VPS_HOST):$(VPS_DIR)/scripts/vps-nginx-install.sh
	@ssh -t $(VPS_USER)@$(VPS_HOST) "sudo bash $(VPS_DIR)/scripts/vps-nginx-install.sh"
	@echo "$(GREEN)✅ Nginx configuré$(NC)"

vps-ssl: ## Obtenir le certificat SSL Let's Encrypt pour $(DOMAIN)
	@echo "$(YELLOW)🔒 Obtention du certificat SSL pour $(DOMAIN)...$(NC)"
	@ssh -t $(VPS_USER)@$(VPS_HOST) "\
		sudo certbot --nginx -d $(DOMAIN) \
		  --non-interactive --agree-tos \
		  --email contact@a3s-securite.com \
		  --redirect \
		  --keep-until-expiring \
	"
	@ssh -t $(VPS_USER)@$(VPS_HOST) "sudo bash $(VPS_DIR)/scripts/vps-nginx-install.sh"
	@echo "$(GREEN)✅ SSL activé sur https://$(DOMAIN)$(NC)"

vps-ssl-renew: ## Tester le renouvellement automatique SSL (dry-run)
	@ssh $(VPS_USER)@$(VPS_HOST) "sudo certbot renew --dry-run"

vps-deploy: ship ## Envoyer le code ET redéployer l'app (ship + deploy)
	@echo "$(YELLOW)🚀 Déploiement de l'application sur le VPS...$(NC)"
	@scp scripts/vps-deploy.sh $(VPS_USER)@$(VPS_HOST):$(VPS_DIR)/scripts/vps-deploy.sh
	@ssh -t $(VPS_USER)@$(VPS_HOST) "cd $(VPS_DIR) && bash scripts/vps-deploy.sh"
	@echo "$(GREEN)✅ Application déployée sur https://$(DOMAIN)$(NC)"

vps-redeploy: ## Re-déployer sans re-envoyer les fichiers (build + PM2 reload)

vps-app-start: ## Démarrer l'app PM2 sur le VPS (premier démarrage)

ship: ## Synchroniser le code source vers le VPS via rsync
	@echo "$(YELLOW)🔄 Synchronisation rsync → $(VPS_USER)@$(VPS_HOST):$(VPS_DIR)...$(NC)"
	@ssh $(VPS_USER)@$(VPS_HOST) "mkdir -p $(VPS_DIR)/scripts $(VPS_DIR)/nginx $(VPS_DIR)/src"
	@rsync -az --delete \
		--exclude='.git/' \
		--exclude='node_modules/' \
		--exclude='dist/' \
		--exclude='logs/' \
		--exclude='uploads/' \
		--exclude='backups/' \
		--exclude='*.log' \
		--exclude='.env.development' \
		--exclude='.env.test' \
		--exclude='package-lock.json' \
		--filter='protect .env.production' \
		$(PWD)/ $(VPS_USER)@$(VPS_HOST):$(VPS_DIR)/
	@echo "$(GREEN)✅ Code synchronisé → $(VPS_USER)@$(VPS_HOST):$(VPS_DIR)$(NC)"

ship-run: ## rsync + build + PM2 reload sur le VPS
	@make ship
	@ssh -t $(VPS_USER)@$(VPS_HOST) "cd $(VPS_DIR) && bash scripts/vps-deploy.sh"
	@echo "$(GREEN)✅ Déployé sur https://$(DOMAIN)$(NC)"

vps-mongo-reinstall: ## Désinstaller complètement MongoDB et réinstaller la version 8.0
	@echo "$(YELLOW)🗑️  Désinstallation complète de MongoDB sur le VPS...$(NC)"
	@ssh -t $(VPS_USER)@$(VPS_HOST) " \
		set -e && \
		echo '=== Arrêt de MongoDB ===' && \
		sudo systemctl stop mongod 2>/dev/null || true && \
		sudo systemctl disable mongod 2>/dev/null || true && \
		echo '=== Purge des paquets ===' && \
		sudo apt-get purge -y mongodb* 2>/dev/null || true && \
		sudo apt-get autoremove -y && \
		echo '=== Suppression des fichiers résiduels ===' && \
		sudo rm -rf /var/lib/mongodb && \
		sudo rm -rf /var/log/mongodb && \
		sudo rm -f /etc/mongod.conf && \
		sudo rm -f /etc/apt/sources.list.d/mongodb*.list && \
		sudo rm -f /usr/share/keyrings/mongodb-server-*.gpg && \
		sudo apt-get update -qq && \
		echo '=== Installation de MongoDB 8.0 ===' && \
		sudo apt-get install -y gnupg curl && \
		curl -fsSL https://www.mongodb.org/static/pgp/server-8.0.asc | sudo gpg --batch --yes -o /usr/share/keyrings/mongodb-server-8.0.gpg --dearmor && \
		echo 'deb [ arch=amd64,arm64 signed-by=/usr/share/keyrings/mongodb-server-8.0.gpg ] https://repo.mongodb.org/apt/ubuntu focal/mongodb-org/8.0 multiverse' | sudo tee /etc/apt/sources.list.d/mongodb-org-8.0.list && \
		sudo apt-get update && \
		sudo apt-get install -y mongodb-org && \
		echo '=== Création de /etc/mongod.conf ===' && \
		printf 'storage:\n  dbPath: /var/lib/mongodb\nsystemLog:\n  destination: file\n  logAppend: true\n  path: /var/log/mongodb/mongod.log\nnet:\n  port: 27017\n  bindIp: 127.0.0.1\nprocessManagement:\n  timeZoneInfo: /usr/share/zoneinfo\n' | sudo tee /etc/mongod.conf && \
		echo '=== Création des répertoires ===' && \
		sudo mkdir -p /var/lib/mongodb /var/log/mongodb && \
		sudo chown -R mongodb:mongodb /var/lib/mongodb /var/log/mongodb && \
		sudo chmod 750 /var/lib/mongodb && \
		echo '=== Démarrage de MongoDB 8.0 ===' && \
		sudo systemctl daemon-reload && \
		sudo systemctl enable mongod && \
		sudo systemctl start mongod && \
		sleep 3 && \
		sudo systemctl status mongod --no-pager | head -15 && \
		echo '=== Test de connexion ===' && \
		mongosh --eval 'db.runCommand({ping:1})' \
	"
	@echo "$(GREEN)✅ MongoDB 8.0 installé et opérationnel$(NC)"

vps-mongo-fix: ## Recréer /etc/mongod.conf manquant et redémarrer MongoDB
	@echo "$(YELLOW)🔧 Correction de MongoDB sur le VPS...$(NC)"
	@ssh -t $(VPS_USER)@$(VPS_HOST) " \
		set -e && \
		echo '--- Création de /etc/mongod.conf ---' && \
		printf 'storage:\n  dbPath: /var/lib/mongodb\nsystemLog:\n  destination: file\n  logAppend: true\n  path: /var/log/mongodb/mongod.log\nnet:\n  port: 27017\n  bindIp: 127.0.0.1\nprocessManagement:\n  timeZoneInfo: /usr/share/zoneinfo\n' | sudo tee /etc/mongod.conf && \
		echo '--- Création des répertoires requis ---' && \
		sudo mkdir -p /var/lib/mongodb /var/log/mongodb && \
		(sudo chown -R mongodb:mongodb /var/lib/mongodb /var/log/mongodb 2>/dev/null || true) && \
		echo '--- Démarrage de MongoDB ---' && \
		sudo systemctl daemon-reload && \
		sudo systemctl enable mongod && \
		sudo systemctl restart mongod && \
		sleep 3 && \
		sudo systemctl status mongod --no-pager | head -20 \
	"
	@echo "$(GREEN)✅ MongoDB corrigé et démarré$(NC)"

vps-mongo-upgrade: ## Mettre à jour MongoDB vers 7.0 sur le VPS (requis : Mongoose 8 nécessite MongoDB 4.2+)	@echo "$(YELLOW)⬆️  Mise à jour MongoDB → 7.0 sur le VPS...$(NC)"
	@ssh -t $(VPS_USER)@$(VPS_HOST) " \
		set -e && \
		echo '--- Arrêt de MongoDB ---' && \
		sudo systemctl stop mongod || true && \
		echo '--- Suppression de l ancienne version ---' && \
		sudo apt-get remove -y mongodb mongodb-clients mongodb-server mongodb-org* 2>/dev/null || true && \
		sudo apt-get purge -y mongodb* 2>/dev/null || true && \
		sudo rm -f /etc/apt/sources.list.d/mongodb*.list && \
		echo '--- Ajout du dépôt MongoDB 7.0 ---' && \
		sudo apt-get install -y gnupg curl && \
		curl -fsSL https://www.mongodb.org/static/pgp/server-7.0.asc | sudo gpg --batch --yes -o /usr/share/keyrings/mongodb-server-7.0.gpg --dearmor && \
		. /etc/os-release && \
		UBUNTU_CODENAME=\$$([ \"\$$VERSION_CODENAME\" = \"jammy\" ] && echo jammy || echo focal) && \
		echo \"deb [ arch=amd64,arm64 signed-by=/usr/share/keyrings/mongodb-server-7.0.gpg ] https://repo.mongodb.org/apt/ubuntu \$$UBUNTU_CODENAME/mongodb-org/7.0 multiverse\" | sudo tee /etc/apt/sources.list.d/mongodb-org-7.0.list && \
		sudo apt-get update && \
		echo '--- Installation de MongoDB 7.0 ---' && \
		sudo apt-get install -y mongodb-org && \
		sudo systemctl daemon-reload && \
		sudo systemctl enable mongod && \
		sudo systemctl start mongod && \
		sleep 2 && \
		echo '--- Version installée ---' && \
		mongod --version \
	"
	@echo "$(GREEN)✅ MongoDB 7.0 installé et démarré sur le VPS$(NC)"

vps-status: ## Vérifier le statut de tous les services sur le VPS
	@echo "$(YELLOW)🔍 Statut des services sur le VPS...$(NC)"
	@ssh $(VPS_USER)@$(VPS_HOST) " \
		echo '--- PM2 ---' && \
		pm2 list 2>/dev/null || echo '  PM2 non démarré' ; \

		node -v 2>/dev/null || echo '  Node.js non installé' ; \
		echo '--- MongoDB ---' && \
		(systemctl is-active mongod 2>/dev/null | grep -q active && echo '  MongoDB: actif' && mongod --version 2>&1 | head -1) || echo '  MongoDB: inactif' ; \
		echo '--- Redis ---' && \
		(systemctl is-active redis-server 2>/dev/null | grep -q active && echo '  Redis: actif' && redis-server --version) || echo '  Redis: inactif' ; \
		echo '--- Nginx ---' && \
		(systemctl is-active nginx 2>/dev/null | grep -q active && echo '  Nginx: actif') || echo '  Nginx: inactif' \
	"
